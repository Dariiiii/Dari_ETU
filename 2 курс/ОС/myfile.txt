Был создан файл my_open.c с кодом внутри:
#include <stdio.h>


int main (int argc, char* argv[]) {
    if (argc > 1) {
        if (fopen(argv[1],"r") == NULL) {
            printf("%s: неудача при попытке открыть файл %s",argv[0],argv[1]);
        }
        else {
            printf("%s: файл %s открыт",argv[0],argv[1]);

    } else {
        printf("открываемый файл не указан");
    }

    return 0;
}

(исправленная первая программа таким образом, чтобы на печать выводилось имя файла),
можно организовать конвейер из двух созданных исполняемых программ командами: || и |

Примеры их работы:
1)$ ./file_name || ./my_open
test_file			// Результат позволяет сделать вывод, что при использовании команды
				|| вывод my_open был перенаправлен в file_name.

2)$ ./file_name | ./my_open
./my_open: файл test_file открыт 	
				// Результат позволяет сделать вывод, что при использовании 
				команды | вывод file_name был перенаправлен в my_open.

Рассмотрим другие варианты конвейеров с помощью операторов &, &&, ;

1)$ ./file_name res && ./my_open test_file
res
./my_open: файл test_file открыт// Оператор && соответствует логическому -И- и используется для запуска нескольких команд.
				Команды выполняются последовательно, и если одна из них не выполняется, тогда и вторая тоже.

2)$ ./file_name res & ./my_open test_file
[1] 208
./my_open: файл test_file открыт
res
[1]+  Done                    ./file_name res
				// Оператор & соответствует логическому -И- и используется для запуска нескольких команд.
				Отличие от && в том, что если одна из команд не выполнится, то вторая может выполнится
				(если в ней нет ошибок).


3)$ ./file_name res ; ./my_open test_file
res
./my_open: файл test_file открыт// Оператор ; просто последовательно выполняет несколько команд вне зависимости от 
				результата выполнения некоторых из них
